140 def SetMyPropertyMap(self, epc:int, epcList:list[int]) -> PDCEDT | None:
141 """!
142 @brief 自身のPropertyMapを設定する
143 @param epc int 0x9d=INF, 0x9e=SET, 0x9f=GET
144 @param epcList list[int]
145 @return PDCEDT | None
146 """
147 # print("SetMyPropertyMap")
148 if epc == 0x9d:
149 self.inf_property_map_raw = epcList
150 elif epc == 0x9e:
151 self.set_property_map_raw = epcList
152 elif epc == 0x9f:
153 self.get_property_map_raw = epcList
154 else:
155 print("ELOBJ Error!! SetMyPropertyMap epc:", hex(epc))
156 return None
157
158 n:int = len(epcList)
159 if n < 16: # format 1
160 pdcedt = PDCEDT()
161 epcList.insert(0, n)
162 pdcedt.setEDT(epcList)
163 self.pdcedts[epc] = pdcedt
164 else: # format 2
165 temp_edt = [0] * 17
166 temp_edt[0] = n
167 for v in epcList:
168 i = (v & 0x0f) + 1
169 flag = 0x01 << ((v >> 4) -8)
170 temp_edt[i] += flag
171 pdcedt = PDCEDT()
172 pdcedt.setEDT(temp_edt)
173 self.pdcedts[epc] = pdcedt
174 return self.pdcedts[epc]
